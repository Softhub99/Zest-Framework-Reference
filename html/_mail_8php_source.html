169  if ((is_array($this->receivers) === false) || (count($this->receivers) < 1)) {

173  if (!empty(trim($this->replyTo))) {

182  $headers[] = 'Cc: '.implode(',', $this->cc);

186  $boundaryContent = md5(rand().'content');
187  //Set the header informations of the mail.

192  $headers[] = 'X-Originating-IP: '.$_SERVER['SERVER_ADDR'];
193  $headers[] = 'Content-Type: multipart/related;boundary='.$boundaryMessage;
194  $headers[] = 'Content-Transfer-Encoding: 8bit';
195  $headers[] = 'From: '.$this->sender;
196  $headers[] = 'Return-Path: '.$this->sender;
197  //Start to generate the content of the mail.
198  $msgContent = "\r\n".'--'.$boundaryMessage."\r\n";
199  $msgContent .= 'Content-Type: multipart/alternative; boundary='.$boundaryContent."\r\n";
200  if (!empty(trim($this->contentPlain))) {

204  }
205  if (!empty(trim($this->contentHTML))) {
